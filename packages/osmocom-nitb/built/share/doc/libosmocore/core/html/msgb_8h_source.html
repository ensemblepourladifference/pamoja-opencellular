453 static inline int msgb_test_invariant(const struct msgb *msg)
454 {
455  const unsigned char *lbound;
456  if (!msg || !msg->data || !msg->tail ||
457  (msg->data + msg->len != msg->tail) ||
458  (msg->data < msg->head) ||
459  (msg->tail > msg->head + msg->data_len))
460  return 0;
461 
462  lbound = msg->head;
463 
464  if (msg->l1h) {
465  if (msg->l1h < lbound)
466  return 0;
467  lbound = msg->l1h;
468  }
469  if (msg->l2h) {
470  if (msg->l2h < lbound)
471  return 0;
472  lbound = msg->l2h;
473  }
474  if (msg->l3h) {
475  if (msg->l3h < lbound)
476  return 0;
477  lbound = msg->l3h;
478  }
479  if (msg->l4h) {
480  if (msg->l4h < lbound)
481  return 0;
482  lbound = msg->l4h;
483  }
484 
485  return lbound <= msg->head + msg->data_len;
486 }
